# リポジトリを新たに作成するコマンド
git init

# Status確認
git status

# ファイルを追加する
git add ***
↓
# localリポジトリにコミットする
git commit -m "add new file"

# リモートリポジトリの情報を追加する
git remote add origin https://github.com//***.git]

# localリポジトリをPushしてリモートリポジトリへ反映する
※masterでうまくいかない場合はmain
git push origin master
